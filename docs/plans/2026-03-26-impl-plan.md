## Task 3: 创建构建脚本

**Files:**
- Create: `scripts/build.js`

**Step 1: 创建构建脚本**

```javascript
#!/usr/bin/env node
/**
 * Clash Config Builder
 * 合并基准配置和自定义规则，生成 substore.js 和 clash.yaml
 */

const fs = require('fs');
const path = require('path');

// 读取规则文件
function readRulesFile(filename) {
  const filePath = path.join(__dirname, '..', 'rules', filename);
  if (!fs.existsSync(filePath)) return [];
  
  const content = fs.readFileSync(filePath, 'utf-8');
  const lines = content.split('\n');
  
  return lines
    .map(line => line.trim())
    .filter(line => line && !line.startsWith('#'))
    .map(line => {
      // 解析格式: RULE,策略组
      const parts = line.split(',');
      if (parts.length >= 2) {
        return {
          rule: parts[0],
          value: parts.slice(1, -1).join(','),
          policy: parts[parts.length - 1]
        };
      }
      return null;
    })
    .filter(Boolean);
}

// 生成 Clash 规则格式
function generateClashRules(rules) {
  return rules.map(r => {
    if (r.rule === 'DOMAIN-SUFFIX') {
      return `DOMAIN-SUFFIX,${r.value},${r.policy}`;
    } else if (r.rule === 'DOMAIN-KEYWORD') {
      return `DOMAIN-KEYWORD,${r.value},${r.policy}`;
    } else if (r.rule === 'IP-CIDR') {
      return `IP-CIDR,${r.value},${r.policy}`;
    } else if (r.rule === 'PROCESS-NAME') {
      return `PROCESS-NAME,${r.value},${r.policy}`;
    }
    return `${r.rule},${r.value},${r.policy}`;
  });
}

// 主函数
function build() {
  console.log('Building clash config...');
  
  // 读取所有规则
  const bypassRules = readRulesFile('bypass.txt');
  const customRules = readRulesFile('custom.txt');
  const proxyRules = readRulesFile('proxy.txt');
  
  console.log(`Loaded rules: bypass=${bypassRules.length}, custom=${customRules.length}, proxy=${proxyRules.length}`);
  
  // 合并规则
  const allExtraRules = [...bypassRules, ...customRules, ...proxyRules];
  const clashRules = generateClashRules(allExtraRules);
  
  // 读取基准配置
  const baseConfigPath = path.join(__dirname, 'base.js');
  let baseConfig = fs.readFileSync(baseConfigPath, 'utf-8');
  
  // 生成规则注入代码
  const rulesInjection = clashRules.map(r => `  - ${r}`).join('\n');
  
  console.log('Generated extra rules:\n' + rulesInjection);
  
  // 输出规则列表（供后续使用）
  const outputPath = path.join(__dirname, '..', 'dist', 'extra-rules.json');
  fs.mkdirSync(path.dirname(outputPath), { recursive: true });
  fs.writeFileSync(outputPath, JSON.stringify({
    bypass: bypassRules,
    custom: customRules,
    proxy: proxyRules,
    clashRules: clashRules
  }, null, 2));
  
  console.log(`Extra rules saved to ${outputPath}`);
}

build();
```

## Task 6: 配置 GitHub Actions

**Files:**
- Modify: `.github/workflows/deploy_clash_config.yml`

**Step 1: 重写工作流**

```yaml
name: Build and Deploy

on:
  push:
    branches:
      - main
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest
    permissions:
      contents: write
    
    steps:
      - name: Checkout
        uses: actions/checkout@v4
      
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
      
      - name: Fetch base config
        run: |
          curl -sL "https://gcore.jsdelivr.net/gh/powerfullz/override-rules@refs/heads/main/convert.min.js" -o scripts/base.js
      
      - name: Build
        run: |
          node scripts/build.js
      
      - name: Deploy to GitHub Pages
        uses: peaceiris/actions-gh-pages@v4
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: ./dist
          force_orphan: true
```
